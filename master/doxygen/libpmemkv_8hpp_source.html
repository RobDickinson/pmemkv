376 template <typename T>
377 inline status config::get_data(const std::string &key, T *&value,
378  std::size_t &count) const noexcept
379 {
380  if (this->_config == nullptr)
381  return status::NOT_FOUND;
382 
383  std::size_t size;
384  auto s = static_cast<status>(pmemkv_config_get_data(
385  this->_config, key.data(), (const void **)&value, &size));
386 
387  if (s != status::OK)
388  return s;
389 
390  count = size / sizeof(T);
391 
392  return status::OK;
393 }

521 {

523  std::min(size(), other.size()));

534 /*
535  * All functions which will be called by C code must be declared as extern "C"
536  * to ensure they have C linkage. It is needed because it is possible that
537  * C and C++ functions use different calling conventions.
538  */
539 extern "C" {
540 static inline int call_get_kv_function(const char *key, size_t keybytes,
541  const char *value, size_t valuebytes, void *arg)
542 {
543  return (*reinterpret_cast<std::function<get_kv_function> *>(arg))(
544  string_view(key, keybytes), string_view(value, valuebytes));
545 }

548 {
549  (*reinterpret_cast<std::function<get_v_function> *>(arg))(
550  string_view(value, valuebytes));
551 }

553 static inline void call_get_copy(const char *v, size_t vb, void *arg)
554 {
